# Определение функции с аннотацией: 
from typing import Dict, Any
def load_config(config_path: str = "config_rpi.yaml") -> Dict[str, Any]:

2. С аннотациями (статическая типизация)
    ✅ Даёт понятные подсказки в IDE (автодополнение)
    ✅ Позволяет проверять типы через mypy/pyright
    ✅ Улучшает читаемость и документированность
    ✅ Помогает избежать ошибок типов
    ❌ Требует Python 3.5+ для typing (современные версии 3.6+)


Dict[str, Any]  # Словарь, где:
# - Ключи всегда строки (str)

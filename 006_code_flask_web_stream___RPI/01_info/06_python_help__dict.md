# Словарь (Dictionary) 
в Python — это коллекция данных, которая хранит пары ключ-значение.
# Это словарь - коллекция пар "ключ-значение"

# Основные характеристики:
1. Структура
python

my_dict = {
    "имя": "Анна",
    "возраст": 25,
    "город": "Москва",
    "хобби": ["чтение", "спорт"]
}

2. Ключи и значения
    Ключи — уникальные идентификаторы (обычно строки, числа или кортежи)
    Значения — любые данные Python (строки, числа, списки, другие словари и т.д.)

3. Доступ к данным
python

# Получить значение по ключу
name = my_dict["имя"]  # "Анна"
age = my_dict["возраст"]  # 25

# Безопасный доступ (чтобы избежать ошибки при отсутствии ключа)
city = my_dict.get("город")  # "Москва"
country = my_dict.get("страна", "Россия")  # "Россия" (значение по умолчанию)

4. Изменение данных
# Добавить новый ключ
my_dict["профессия"] = "инженер"

# Изменить значение
my_dict["возраст"] = 26

# Удалить ключ
del my_dict["город"]

5. Проверка наличия ключа
python

if "возраст" in my_dict:
    print("Ключ 'возраст' существует")

6. Итерация (перебор)
# Только ключи
for key in my_dict:
    print(key)

# Ключи и значения
for key, value in my_dict.items():
    print(f"{key}: {value}")

# Только значения
for value in my_dict.values():
    print(value)

7. Пример конфигурации (как в вашем коде)
python

config = {
    "database": {
        "host": "localhost",
        "port": 5432,
        "name": "my_db"
    },
    "server": {
        "port": 8000,
        "debug": True
    }
}

# Доступ к вложенным значениям
db_host = config["database"]["host"]  # "localhost"
server_port = config["server"]["port"]  # 8000

8. Особенности

    Ключи должны быть неизменяемыми (строки, числа, кортежи)

    Ключи уникальны — нельзя иметь два одинаковых ключа

    Не сохраняет порядок (в Python 3.7+ сохраняет порядок добавления)

    Быстрый поиск по ключу — O(1) в среднем случае

9. Полезные методы
# Количество пар ключ-значение
count = len(my_dict)  # 3

# Все ключи
keys = my_dict.keys()  # dict_keys(['имя', 'возраст', 'хобби'])

# Все значения
values = my_dict.values()  # dict_values(['Анна', 26, ['чтение', 'спорт']])

# Очистить словарь
my_dict.clear()

def load_config() -> Dict[str, Any]:
    return {
        "server": {
            "host": "localhost",  # это строка (str)
            "port": 8080          # это целое число (int), НЕ строка!
        },
        "database": {
            "name": "test",
            "user": "admin"
        }
    }

config = load_config()

# Правильно:
print(config["server"]["host"])     # "localhost" (str)
print(config["server"]["port"])     # 8080 (int) - БЕЗ кавычек!

# Неправильно было бы:
print(config["server"]["port"])     # "8080"  <-- С кавычками - это уже строка
